1)QUERY OF PLSQL

DECLARE
V_ENAME VARCHAR2(10);
V_EID EMPLOYEE.EMP_ID%TYPE:=&EMP_ID;
BEGIN
SELECT EMP_NAME INTO V_ENAME
FROM EMPLOYEE
WHERE EMP_ID=V_EID;
DBMS_OUTPUT.PUT_LINE('THE EMPLOYEE NAME FOR EMPLOYEE ID'||V_EID||'IS'||V_ENAME);
END;
/
2)QUERY USING COMPLEX VARIABLE

DECLARE
V_RECNAME EMPLOYEE%ROWTYPE;
BEGIN
SELECT * INTO V_RECNAME
FROM employee
WHERE EMP_ID=&EMP_ID;
DBMS_OUTPUT.PUT_LINE('THE EMPLOYEE NAME IS'||V_RECNAME.EMP_NAME);
DBMS_OUTPUT.PUT_LINE('THE SALARY IS'||V_RECNAME.SALARY);
END;
/
3)QUERY TO USE GLOBAL DECLARATION

  VARIABLE EID NUMBER;

4)QUERY USING CURSOR TO DISPLAY NUMBER OF ROWS AFFECTED

DECLARE
V_EMP_NAME VARCHAR2(30):='&EMP_NAME';
V_EMP_ID NUMBER :=&EMP_ID;
V_ROWCOUNT V_EMP_ID%TYPE;
BEGIN
      UPDATE EMPLOYEE
      SET EMP_NAME=V_EMP_NAME
      WHERE EMP_ID=V_EMP_ID;
      V_ROWCOUNT :=SQL%ROWCOUNT;
      DBMS_OUTPUT.PUT_LINE('THE NUMBER OF ROWS AFFECTED IS '||V_ROWCOUNT);
      END;
      /
